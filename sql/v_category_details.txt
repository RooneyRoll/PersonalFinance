
CREATE OR REPLACE VIEW public.v_category_budgets AS
 SELECT mas.id,
    mas.amount,
    mas.from_date,
    mas.to_date,
    det.pcat_name AS cname,
    det.pcat_user AS user_id
   FROM category_budget mas
     JOIN payment_categories det ON mas.category_id = det.pcat_id;

ALTER TABLE public.v_category_budgets
    OWNER TO postgres;
